      // 封装二叉搜索树
      function BinarySearchTree() {
        // 节点
        function Node(key, left, right) {
          this.key = key;
          this.right = null;
          this.left = null;
        }
        // 初始化根节点
        this.root = null;
        // 方法
        // 插入数据
        BinarySearchTree.prototype.insert = function (key) {
          // 根据key创建节点
          let newNode = new Node(key);
          // 判断根节点是否有值
          if (this.root == null) {
            this.root = newNode;
          } else {
            this.insertNode(this.root, newNode);
          }
        };
        // 插入节点的递归调用函数
        BinarySearchTree.prototype.insertNode = function (node, newNode) {
          if (node.key > newNode.key) {
            // 向左查找
            if (node.left == null) {
              node.left = newNode;
            } else {
              this.insertNode(node.left, newNode);
            }
          } else {
            // 向右查找
            if (node.right == null) {
              node.right = newNode;
            } else {
              this.insertNode(node.right, newNode);
            }
          }
        };
        // 先序遍历
        BinarySearchTree.prototype.preOrderTraversal = function (handler) {
          this.preOrderTraversalNode(this.root, handler);
        };
        // 先序遍历的递归调用方法
        BinarySearchTree.prototype.preOrderTraversalNode = function (
          node,
          handler
        ) {
          if (node != null) {
            // 处理经过的节点
            handler(node.key);
            // 处理经过节点的左子节点
            this.preOrderTraversalNode(node.left, handler);
            // 处理经过节点的右子节点
            this.preOrderTraversalNode(node.right, handler);
          }
        };
        // 先序遍历 -- 迭代
        BinarySearchTree.prototype.preOrderTraversalIteration = function () {
          let node = this.root;
          let result = [];
          let stack = [];
          if (this.root == null) {
            return [];
          }
          stack.push(node);
          while (stack.length !== 0) {
            node = stack.pop();
            result.push(node.key);
            if (node.right !== null) {
              stack.push(node.right);
            }
            if (node.left !== null) {
              stack.push(node.left);
            }
          }
          return result;
        };
        // 中序遍历
        BinarySearchTree.prototype.midOrderTraversal = function (handler) {
          this.midOrderTraversalNode(this.root, handler);
        };
        // 中序遍历的递归调用方法
        BinarySearchTree.prototype.midOrderTraversalNode = function (
          node,
          handler
        ) {
          if (node != null) {
            // 处理经过节点的左子节点
            this.midOrderTraversalNode(node.left, handler);
            // 处理经过的节点
            handler(node.key);
            // 处理经过节点的右子节点
            this.midOrderTraversalNode(node.right, handler);
          }
        };
        // 中序遍历非递归法 -- 迭代
        BinarySearchTree.prototype.midOrderTraversalIteration = function () {
          let stack = [];
          let result = [];
          let node = this.root;

          while (node != null || stack.length != 0) {
            while (node != null) {
              stack.push(node);
              node = node.left;
            }
            node = stack.pop();
            result.push(node.key);
            node = node.right;
          }
          return result;
        };
        // 后序遍历
        BinarySearchTree.prototype.postOrderTraversal = function (handler) {
          this.postOrderTraversalNode(this.root, handler);
        };
        BinarySearchTree.prototype.postOrderTraversalNode = function (
          node,
          handler
        ) {
          if (node != null) {
            // 处理经过节点的左子节点
            this.postOrderTraversalNode(node.left, handler);
            // 处理经过节点的右子节点
            this.postOrderTraversalNode(node.right, handler);
            // 处理经过的节点
            handler(node.key);
          }
        };
        // 后序遍历 -- 迭代
        BinarySearchTree.prototype.postOrderTraversalIteration = function () {
          let result = [];
          let stack = [];
          if (this.root == null) {
            return [];
          }
          let node = this.root;
          stack.push(node);
          while (stack.length !== 0) {
            node = stack.pop();
            result.unshift(node.val);
            if (node.left !== null) {
              stack.push(node.left);
            }
            if (node.right !== null) {
              stack.push(node.right);
            }
          }
          return result;
        };
        // 层序遍历
        BinarySearchTree.prototype.levelOrderTraversal = function (handler) {
          this.levelOrderTraversalNode(this.root, handler);
        };
        BinarySearchTree.prototype.levelOrderTraversalNode = function (
          node,
          handler
        ) {
          let queue = []; // 使用数组模拟队列
          queue.push(node);
          while (queue.length) {
            node = queue.shift();
            handler(node.key);
            if (node.left != null) {
              queue.push(node.left);
            }
            if (node.right != null) {
              queue.push(node.right);
            }
          }
        };
        // 获取最小值
        BinarySearchTree.prototype.min = function () {
          // 获取根节点
          let node = this.root;
          // 依次向左不断查找，直到节点为null
          while (node.left != null) {
            node = node.left;
          }
          return node.key;
        };
        // 获取最大值
        BinarySearchTree.prototype.max = function () {
          // 获取根节点
          let node = this.root;
          // 依次向右不断查找，直到节点为null
          while (node.right != null) {
            node = node.right;
          }
          return node.key;
        };
        // 搜索特定的key值
        // key值存在返回true,不存在返回false
        BinarySearchTree.prototype.search = function (key) {
          return this.searchNode(this.root, key);
        };
        BinarySearchTree.prototype.searchNode = function (node, key) {
          // 传入的node为null就退出递归
          if (node === null) {
            return false;
          }
          // 判断node节点的值和传入的key值
          if (node.key > key) {
            // 向左查找
            return this.searchNode(node.left, key);
          } else if (node.key < key) {
            // 向右查找
            return this.searchNode(node.right, key);
          } else {
            return true;
          }
        };

        // 删除节点
        BinarySearchTree.prototype.remove = function (key) {
          // 1. 寻找要删除的节点，看是否存在该节点
          // 1.1 定义变量，保存信息
          let current = this.root; // 要删除的节点，初始化为根节点
          let parent = null; // 要删除的节点的父节点,初始化为null
          let isLeftChild = true; // 要删除的是左节点还是右节点

          //2. 开始寻找删除的节点
          while (current.key != key) {
            parent = current;
            if (key < current.key) {
              isLeftChild = true;
              current = current.left;
            } else {
              isLeftChild = false;
              current = current.right;
            }
            // 没有找到
            if (current == null) return false;
          }
          // 3. 根据情况进行删除
          // 3.1 删除的节点是叶子节点
          if (current.left == null && current.right == null) {
            if (current == this.root) {
              this.root = null;
            } else if (isLeftChild) {
              parent.left = null;
            } else {
              parent.right = null;
            }
          }
          // 3.1 删除的节点有一个子节点
          else if (current.right == null) {
            // 要删除节点的右节点为空
            if (current == this.root) {
              this.root = current.left;
            } else if (isLeftChild) {
              parent.left = current.left;
            } else {
              parent.right = current.left;
            }
          } else if (parent.left == null) {
            // 要删除节点的左节点为空
            if (current == this.root) {
              this.root = current.right;
            } else if (isLeftChild) {
              parent.left = current.right;
            } else {
              parent.right = current.right;
            }
          }
          // 3.3 删除的节点有两个子节点
          else {
            // 获取后继节点
            let success = this.getSuccess(current);
            // 判断删除节点是否为根节点
            if (current == this.root) {
              this.root = success;
            } else if (isLeftChild) {
              parent.left = success;
            } else {
              parent.right = success;
            }
            // 将删除节点的左子树 = current.left
            success.left = current.left;
          }
        };
        // 找后继
        BinarySearchTree.prototype.getSuccess = function (delNode) {
          // 定义变量，保存找到的后继
          let success = delNode;
          let current = delNode.right; // 后继在要删除节点的右子树中,即右子树中的最小值
          let successParent = null; // 后继节点的父节点
          // 循环查找
          while (current != null) {
            successParent = success;
            success = current;
            current = current.left;
          }
          // 判断寻找到的后继节点是否就是删除节点的right节点
          if (success != delNode.right) {
            successParent.left = success.right;
            success.right = delNode.right;
          }
          return success;
        };
      }
